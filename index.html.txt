  <title>Кликер</title>
  <style>
    body {
      font-family: Arial, sans-serif;
      text-align: center;
      padding-top: 50px;
      background-color: #f0f0f0;
    }
    button {
      transition: transform 0.1s ease;
    }
    .clicked {
      transform: scale(1.2);
    }
    #clickButton {
      font-size: 24px;
      padding: 20px 40px;
      margin: 20px;
    }
    #upgradeButton {
      padding: 10px 20px;
      font-size: 18px;
    }
    .info {
      font-size: 20px;
      margin: 10px;
    }
  </style>

<body>
  <h1>Кликер</h1>
  <div class="info">Очки: <span id="score">0</span></div>
  <button id="clickButton">Кликни!</button>
  <div class="info">Сила клика: <span id="clickPower">1</span></div>
  <button id="upgradeButton">Улучшить (10 очков)</button>

  <script>
    let score = 0;
    let clickPower = 1;

    const scoreEl = document.getElementById('score');
    const clickPowerEl = document.getElementById('clickPower');
    const clickButton = document.getElementById('clickButton');
    const upgradeButton = document.getElementById('upgradeButton');

    function formatNumber(num) {
      return num.toLocaleString('en-US');
    }

    function saveData() {
      const data = {
        score: score,
        clickPower: clickPower,
      };
      localStorage.setItem('clickerSave', JSON.stringify(data));
    }

    function loadData() {
      const data = localStorage.getItem('clickerSave');
      if (data) {
        const parsed = JSON.parse(data);
        score = parsed.score || 0;
        clickPower = parsed.clickPower || 1;
      }
    }

    function animateButton(button) {
      button.classList.add('clicked');
      setTimeout(() => {
        button.classList.remove('clicked');
      }, 100);
    }

    function updateUI() {
      scoreEl.textContent = formatNumber(score);
      clickPowerEl.textContent = formatNumber(clickPower);
    }

    clickButton.addEventListener('click', () => {
      score += clickPower;
      updateUI();
      animateButton(clickButton);
      saveData();
    });

    upgradeButton.addEventListener('click', () => {
      animateButton(upgradeButton);
      if (score >= 10) {
        score -= 10;
        clickPower++;
        updateUI();
        saveData();
      } else {
        alert('Недостаточно очков для улучшения!');
      }
    });

    loadData();
    updateUI();
  </script>
</body>
